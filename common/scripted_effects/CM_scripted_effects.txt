########## Construction Management Panel Scripted Effects ##########

CM_clear_state_resources_infastructure_objects = {
	clear_variable = CM_state_sum_of_infrastructure_bonus_resources
	clear_variable = CM_state_infrastructure_construction_speed
	
	clear_array = CM_state_resources_coefficients
}

CM_clear_possible_construction_targets_objects = {
	for_each_scope_loop = {
		array = CM_possible_construction_targets
		
		clear_variable = CM_state_ownership_frame
		clear_variable = CM_state_buildings_construction_speed
		clear_variable = CM_state_industrial_complex_construction_speed
		clear_variable = CM_state_arms_factory_construction_speed
		clear_variable = CM_state_naval_dockyard_construction_speed
		
		CM_clear_state_resources_infastructure_objects = yes
	}
	
	clear_array = CM_possible_construction_targets
	
	clear_variable = CM_states_names_panel_frame
	clear_variable = CM_state_construction_speed_panel_frame
	clear_variable = CM_state_industrial_complex_construction_speed_panel_frame
	clear_variable = CM_state_arms_factory_construction_speed_panel_frame
	clear_variable = CM_state_naval_dockyard_construction_speed_panel_frame
}

CM_find_possible_construction_targets_last_temp_index = {
	set_temp_variable = { CM_possible_construction_targets_last_temp_index = CM_possible_construction_targets^num }
	subtract_from_temp_variable = { CM_possible_construction_targets_last_temp_index = 1 }
}
CM_set_basic_possible_construction_targets_objects_for_priority_reconstruction = {
	set_temp_variable = { CM_highest_priority_possible_construction_target_temp_index = cm_current_possible_construction_target_temp_index }
	
	set_temp_variable = { CM_next_possible_construction_target_temp_index = cm_current_possible_construction_target_temp_index }
	add_to_temp_variable = { CM_next_possible_construction_target_temp_index = 1 }
}

CM_swap_possible_construction_targets_if_needed = {
	if = {
		limit = {
			check_variable = {
				var = cm_current_possible_construction_target_temp_index
				compare = not_equals
				value = CM_highest_priority_possible_construction_target_temp_index
			}
		}
		
		set_temp_variable = { CM_current_possible_construction_targets_temp_value = CM_possible_construction_targets^cm_current_possible_construction_target_temp_index }
		set_variable = { CM_possible_construction_targets^cm_current_possible_construction_target_temp_index = CM_possible_construction_targets^CM_highest_priority_possible_construction_target_temp_index }
		set_variable = { CM_possible_construction_targets^CM_highest_priority_possible_construction_target_temp_index = CM_current_possible_construction_targets_temp_value }
	}
}

CM_complete_calculation_of_state_steel_coefficient = {

	multiply_variable = { CM_state_resources_coefficients^0 = CM_state_infrastructure_temp_impact_on_resources?CM_state_infrastructure_construction_speed }
	round_variable = CM_state_resources_coefficients^0
	divide_variable = { CM_state_resources_coefficients^0 = 10 }

CM_reconstruct_possible_construction_targets_array_for_descending_state_and_terrain_strategic_value = {
	CM_find_possible_construction_targets_last_temp_index = yes
	
	for_loop_effect = {
		start = 0
		end = CM_possible_construction_targets_last_temp_index
		value = cm_current_possible_construction_target_temp_index
		
		CM_set_basic_possible_construction_targets_objects_for_priority_reconstruction = yes
		
		for_loop_effect = {
			start = CM_next_possible_construction_target_temp_index
			compare = less_than_or_equals
			end = CM_possible_construction_targets_last_temp_index
			
			var:CM_possible_construction_targets^CM_highest_priority_possible_construction_target_temp_index = {
				var:ROOT.CM_possible_construction_targets^v = {
					if = {
						limit = {
							check_variable = { state_and_terrain_strategic_value > PREV.state_and_terrain_strategic_value }
						}
						
						set_temp_variable = { CM_highest_priority_possible_construction_target_temp_index = v }
					}
				}
			}
		}
		
		CM_swap_possible_construction_targets_if_needed = yes
	}
	
	set_variable = { CM_states_names_panel_frame = 2 }
}

	add_to_temp_variable = { CM_temp_country_resources_multiplier = modifier@local_resources_factor }

		limit = { check_variable = { CM_infrastructure_construction_targets^num > 0 } }

			array = CM_infrastructure_construction_targets
			
			CM_adjust_country_resources_array = yes
		}
	}
	
	if = {
		limit = { CM_there_is_no_shortage_of_resources = yes }
		
		set_country_flag = CM_imported_resources_has_been_removed_flag
		
		subtract_from_temp_variable = { CM_temp_country_resources_array^0 = resource_imported@steel }
		subtract_from_temp_variable = { CM_temp_country_resources_array^1 = resource_imported@crystals }
		subtract_from_temp_variable = { CM_temp_country_resources_array^2 = resource_imported@tungsten }
		subtract_from_temp_variable = { CM_temp_country_resources_array^3 = resource_imported@aluminium }
		subtract_from_temp_variable = { CM_temp_country_resources_array^4 = resource_imported@rubber }
		subtract_from_temp_variable = { CM_temp_country_resources_array^5 = resource_imported@chromium }
		subtract_from_temp_variable = { CM_temp_country_resources_array^6 = resource_imported@oil }
		
		if = {
			limit = { CM_there_is_no_shortage_of_resources = yes }
			
			set_country_flag = CM_consumed_resources_has_been_added_flag
			set_country_flag = CM_non_consumed_resources_has_been_added_flag
			
			if = {
				limit = { check_variable = { resource_consumed@steel = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^0 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { check_variable = { resource_consumed@crystals = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^1 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { check_variable = { resource_consumed@tungsten = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^2 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { check_variable = { resource_consumed@aluminium = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^3 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { check_variable = { resource_consumed@rubber = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^4 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { check_variable = { resource_consumed@chromium = 0 } }
				add_to_temp_variable = { CM_temp_country_resources_array^5 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
			if = {
				limit = { NOT = { fuel_ratio < 0.95 } }
				add_to_temp_variable = { CM_temp_country_resources_array^6 = 10000 }
				clr_country_flag = CM_non_consumed_resources_has_been_added_flag
			}
		}
	}
	
	if = {
		limit = { check_variable = { CM_states_filter_regime?1 = 1 } }
		CM_create_cored_owned_controlled_states_array_for_infrastructure_construction = yes
	}
	else_if = {
		limit = { check_variable = { CM_states_filter_regime = 3 } }
		CM_create_owned_controlled_states_array_for_infrastructure_construction = yes
	}
	else = { # CM_states_filter_regime = 2
		CM_create_controlled_states_array_for_infrastructure_construction = yes
	}
}

		if = {

		}

CM_check_condition_for_the_next_industrial_complexes_construction_iteration = {
	if = {
		limit = { check_variable = { CM_temp_num_of_possible_free_building_slots = 1 } }
		set_temp_variable = { cm_while_loop_breaker = 1 }
	}
	else = {
		subtract_from_temp_variable = { CM_temp_num_of_possible_free_building_slots = 1 }
		
		if = {
			limit = {
				check_variable = {
					var = CM_temp_num_of_civilian_factories_available_for_construction_line
					compare = not_equals
					value = 15
				}
			}
			
			add_to_temp_variable = { CM_temp_proportion_of_unavailable_industrial_complex = ROOT.modifier@consumer_goods_factor }
			
			if = {
				limit = { check_variable = { CM_temp_proportion_of_unavailable_industrial_complex > 1 } }
				subtract_from_temp_variable = { CM_temp_proportion_of_unavailable_industrial_complex = 1 }
			}
			else = {
				add_to_temp_variable = { CM_temp_num_of_civilian_factories_available_for_construction_line = 1 }
			}
		}
	}
}

		limit = { check_variable = { CM_temp_num_of_possible_free_building_slots = 1 } }

	}
	else = {
		subtract_from_temp_variable = { CM_temp_num_of_possible_free_building_slots = 1 }
		add_to_temp_variable = { CM_temp_proportion_of_unavailable_industrial_complex = ROOT.modifier@consumer_goods_factor }

			limit = { check_variable = { CM_temp_proportion_of_unavailable_industrial_complex > 1 } }

			subtract_from_temp_variable = { CM_temp_possible_num_of_civilian_factories_available_for_projects = 1 }

				limit = { check_variable = { CM_temp_possible_num_of_civilian_factories_available_for_projects < 15 } }

	}

CM_add_naval_dockyards_and_infrastructure_constructions = {
	if = {
		limit = { check_variable = { CM_states_filter_regime?1 = 1 } }
		
		if = {
			limit = { check_variable = { CM_ROI_coefficient?1 = 1 } }
			
			if = {
				limit = { check_variable = { CM_temp_possible_num_of_civilian_factories_available_for_projects < 15 } }
				
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_fastest_ROI = yes
					}
				}
			}
			else = {
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_fastest_ROI_full_line = yes
					}
				}
			}
		}
		else_if = {
			limit = { check_variable = { CM_ROI_coefficient = 2 } }
			
			if = {
				limit = { check_variable = { CM_temp_possible_num_of_civilian_factories_available_for_projects < 15 } }
				
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_optimal_ROI = yes
					}
				}
			}
			else = {
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_optimal_ROI_full_line = yes
					}
				}
			}
		}
		else = { # CM_ROI_coefficient = 3
			CM_calculate_temp_min_num_of_infrastructure_for_construction = yes
			
			if = {
				limit = { check_variable = { CM_temp_possible_num_of_civilian_factories_available_for_projects < 15 } }
				
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_maximum_ROI = yes
					}
				}
			}
			else = {
				for_each_scope_loop = {
					array = core_states
					
					if = {
						limit = { CM_is_owned_and_controlled_with_free_building_slot_for_naval_dockyard_and_not_blocked = yes }
						CM_calculate_state_naval_dockyards_and_infrastructure_construction_objects_with_num_of_free_IC_slots_for_maximum_ROI_full_line = yes
					}
				}
			}
		}
	}
	else_if = {
		limit = { check_variable = { CM_states_filter_regime = 3 } }
		CM_create_owned_controlled_states_array_for_naval_dockyards_with_infrastructure_construction = yes
	}
	else = { # CM_states_filter_regime = 2
		CM_create_controlled_states_array_for_naval_dockyards_and_infrastructure_construction = yes
	}
	
	#print_variables = {
	#	file = CM_log_file
	#	text = CM_country_naval_dockyards_calculation_variables
	#	append = yes
	#}
	
	while_loop_effect = {
		limit = { always = yes } # do_while
		
		random_scope_in_array = {
			array = CM_temp_possible_construction_targets
			limit = { CM_has_highest_priority_for_buildings_and_infrastructure_construction = yes }
			
			if = {
				limit = { check_variable = { CM_state_num_of_infrastructure_for_construction > 0 } }
				CM_reconstruct_infrastructure_costructions = yes
			}
			else = {
				add_building_construction = {
					type = dockyard
					level = 1
					instant_build = no
				}
			}
			
			CM_remove_from_possible_construction_targets_array = yes
		}
		
		if = {
			limit = { CM_no_available_civilian_factories_OR_no_possible_construction_sites = yes }
			set_temp_variable = { break = 1 }
		}
		else_if = {
			limit = { CM_is_new_possible_construction_targets_array = yes }
			
			if = {
				limit = { check_variable = { CM_states_filter_regime?1 = 1 } }
				
				if = {
					limit = { has_country_flag = CM_owned_controlled_states_has_been_added_flag }
					
					CM_create_controlled_states_array_for_naval_dockyards_and_infrastructure_construction = yes
					CM_decrease_added_targets_scopes_flags = yes
				}
				else = {
					CM_create_owned_controlled_states_array_for_naval_dockyards_with_infrastructure_construction = yes
					set_country_flag = CM_owned_controlled_states_has_been_added_flag
				}
			}
			else = { # CM_states_filter_regime = 3
				CM_create_controlled_states_array_for_naval_dockyards_and_infrastructure_construction = yes
				set_country_flag = CM_controlled_states_has_been_added_flag
			}
		}
	}
	
	CM_clear_temp_possible_buildings_and_infrastructure_construction_targets_objects = yes
	CM_update_possible_construction_targets_array_if_needed = yes
}
